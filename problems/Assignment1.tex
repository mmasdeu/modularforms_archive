\section{The slash operator}

Let $\gamma$ and $\gamma'$ be in $\GL_2^+(\RR)$, $k$ and $\ell$ integers, and let $f,g\colon \HH\to \CC$ be arbitrary functions. Define
\[
f|_k\gamma = (\det\gamma)^{k-1}j(\gamma,z)^{-k} f(\gamma z).
\]
\begin{enumerate}
\item Show that:
\[
(f|_k\gamma)|_k\gamma' = f|_k(\gamma\gamma').
\]
\item Show that, if $\gamma$ is in $\SL_2(\RR)$, then
\[
(f|_k\gamma)(g|_\ell \gamma)=(fg)|_{k+\ell}\gamma.
\]
\end{enumerate}

\section{The divisor function}

Recall that we have defined, for $r\geq 0$ and $n\geq 1$,
\[
\sigma_{r}(n)=\sum_{d\mid n} d^{r},
\]
where the sum runs over all positive divisors $d$ of $n$.
\begin{enumerate}
\item Suppose that $m$ and $n$ are coprime. Show that
\[
\sigma_{r}(mn)=\sigma_{r}(m)\sigma_{r}(n).
\]
In number theory, such a function is called ``multiplicative''.
\item Let $n=\prod_p p^{e_p}$ be the prime factorization of $n$. Prove that
\[
\sigma_{r}(n)=\prod_p \frac{p^{r(e_p+1)} -1 }{p^{r}-1}.
\]
\end{enumerate}

\section{Integrality of the Delta function}

Define $\Delta$ in terms of $E_4$ and $E_6$ as follows:
\[
\Delta = \frac{E_4^3-E_6^2}{1728}.
\]
Show that if $\Delta(q)=\sum \tau(n)q^n$, then $\tau(n)\in\ZZ$ for all $n\geq 1$. Do not use the product formula for $\Delta$.


\section{Invariant functions from modular forms}

Let $f$ and $g$ be modular forms of weight $k$.
\begin{enumerate}
\item Show that the function
\[
\varphi\colon \HH\to \CC,\quad z\mapsto f(z)\ol{g(z)}(\Im z)^k
\]
is $\SL_2(\ZZ)$-invariant. That is, $\varphi(\gamma z)=\varphi(z)$ for all $\gamma\in\SL_2(\ZZ)$.
\item Suppose that either $f$ or $g$ is a cusp form. Show that the function $z\mapsto |\varphi(z)|$ is bounded on $\HH$.
\end{enumerate}


\section{Bernoulli numbers}
Define the Bernoulli numbers $B_k$ by the formal power series expansion
\[
\frac{t}{e^t-1} = \sum_{k=0}^\infty B_k\frac{t^k}{k!}.
\]
\begin{enumerate}
\item Show that $B_1 = -1/2$ and $B_k=0$ for all other odd $k$.
\item Show that $B_2=1/6$, $B_4=-1/30$ and $B_6=1/42$.
\item Let $\zeta(k) = \sum_{n=1}^\infty n^{-k}$. Show that for all even $k\geq 2$ we have
\[
2\zeta(k) = -\frac{(2\pi i)^k}{k!}B_k.
\]
You can use without proof
the formula (see Lemma~1.4.4 in the notes)
\[
\frac{1}{z} + \sum_{d=1}^\infty \left(\frac{1}{z-d} +\frac{1}{z+d}\right) = \pi i-2\pi i\sum_{m=0}^\infty e^{2\pi i m z},\quad z\in \HH.
\]
In particular, we obtain the well known formulas $\zeta(2) = \pi^2/6$, $\zeta(4) = \pi^4/90$, $\zeta(6) = \pi^6/945$, \ldots
\end{enumerate}
